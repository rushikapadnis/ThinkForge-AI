# ── Core Web Framework ─────────────────────────────────────────────

# FastAPI is the main framework used to build APIs.
# Why we need it: It lets us create routes like /research and /health easily and efficiently.
fastapi==0.115.5

# Uvicorn is the server that actually runs the FastAPI app.
# Why we need it: FastAPI is just code, uvicorn makes it a live web server.
# [standard] adds extra performance improvements and production-ready features.
uvicorn[standard]==0.32.0


# ── AI / Agent Framework ────────────────────────────────────────────

# LangGraph is used to build multi-step AI workflows (agent graphs).
# Why we need it: It allows us to chain Planner → Researcher → Writer → Reviewer like a pipeline.
langgraph==0.2.55

# LangChain is the core framework for working with LLMs.
# Why we need it: It provides tools to connect prompts, models, and logic together.
langchain==0.3.9

# LangChain integration for Groq models.
# Why we need it: So we can use Groq LLaMA models inside LangChain easily.
langchain-groq==0.2.4

# LangChain integration for OpenAI models.
# Why we include it: Optional support for GPT models if we switch providers.
langchain-openai==0.2.10

# Community extensions for LangChain.
# Why we need it: Provides extra tools like document loaders, utilities, and helpers.
langchain-community==0.3.9
